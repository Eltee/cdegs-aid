122 int const& Energization::getRealPart() const{
123  if(m_angle != 0){
124  int result = static_cast<int>((m_magnitude * std::cos(m_angle * PI / 180)));
125  if(result > 0){
126  return result + 1;
127  }
128  else{
129  return result - 1;
130  }
131  }
132  else{
133  return m_magnitude;
134  }
135 }
