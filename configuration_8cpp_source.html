323 Configuration& Configuration::removeTolerance(double const& tolerance){
324  bool done;
325 
326  do{
327  done = true;
328  for(std::vector<double>::iterator it=m_tolerances.begin(); it!=m_tolerances.end(); it++){
329  if(*it == tolerance){
330  m_tolerances.erase(it);
331  done = false;
332  break;
333  }
334  }
335  }while(!done);
336 
337  return *this;
338 }

733 int Configuration::idGenerator(const std::string& type){
734  int result = 0;
735 
736  if(type == "LeadTypeId"){
737  result = m_ids.leadTypeId;
738  m_ids.leadTypeId += 1;
739  }
740  else if(type == "CoatingId"){
741  result = m_ids.coatingId;
742  m_ids.coatingId += 1;
743  }
744  else if(type == "EnergizationId"){
745  result = m_ids.energizationId;
746  m_ids.energizationId += 1;
747  }
748  else if(type == "ConductorTypeId"){
749  result = m_ids.conductorTypeId;
750  m_ids.conductorTypeId += 1;
751  }
752  else if(type == "CableTypeId"){
753  result = m_ids.cableTypeId;
754  m_ids.cableTypeId += 1;
755  }
756  else if(type == "ConductorId"){
757  result = m_ids.conductorId;
758  m_ids.conductorId += 1;
759  }
760  else if(type == "BuildingConductorId"){
761  result = m_ids.buildingConductorId;
762  m_ids.buildingConductorId += 1;
763  }
764  else if(type == "BuildingId"){
765  result = m_ids.buildingId;
766  m_ids.buildingId += 1;
767  }
768  else if(type == "ProfileId"){
769  result = m_ids.profileId;
